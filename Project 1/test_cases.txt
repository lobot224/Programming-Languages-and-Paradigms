# Nested Operations
(+ (* 2 3) (- 10 4))
(- (/ 15 3) (* 2 2))
(+ (/ (+ 12 3) (* 2 2)) 5)

# Valid History References
(+ 5 10)
(* $1 2)
(- $2 $1)
(/ $3 5)

# Floating Point and Precision
(/ 7.5 2.5)
(* $1 2.1)
(+ $2 (- 5.6 1.1))

# Division by Zero
(/ 10 0)
(/ $1 0)

# Invalid Input
(+ 5 invalid)
(- $100 3)
(+ $2 $-3)

# Unary Operations
(- 5)
(- (* 2 3))
(+ (- 4) (- (- 10)))

# Complex Nested Expressions
(+ 5 7)
(* $1 3)
(/ (+ $1 $2) (- 20 10))
(+ (* $3 $1) (/ $2 $3))

# Deeply Nested Expressions
(+ (* (- (/ 16 4) 2) 5) (/ (* 6 3) (+ 1 2)))
